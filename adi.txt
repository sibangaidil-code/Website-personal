    <script>
        function login() {
            const usernameInput = document.getElementById('username').value;
            const passwordInput = document.getElementById('password').value;
            const messageElement = document.getElementById('message');
            const loginContainer = document.getElementById('login-form-container');
            const successContent = document.getElementById('success-content');

            // Kredensial yang Benar
            const correctUsername = "cekbi";
            const correctPassword = "kaksah";

            if (usernameInput === correctUsername && passwordInput === correctPassword) {
                // Login Berhasil
                loginContainer.style.display = 'none'; // Sembunyikan form
                successContent.style.display = 'block'; // Tampilkan konten sukses
                // Mengosongkan pesan error
                messageElement.textContent = ""; 
            } else {
                // Login Gagal
                messageElement.textContent = "Username atau Password salah! Coba lagi.";
            }
        }
    </script>
